## Installation

```npm tab="React"
npm i @docukit/docsync-react @docukit/docnode
```

```npm tab="Vanilla"
npm i @docukit/docsync @docukit/docnode
```

`@docukit/docnode` is required when using `DocNodeBinding`. DocSync can support other document bindings, so DocNode is an optional peer dependency instead of a hard runtime dependency.

## 1. Client

```ts tab="React"
// client.ts
import {
  createDocSyncClient,
  indexedDBProvider,
} from "@docukit/docsync-react/client";
import { DocNodeBinding } from "@docukit/docsync-react/docnode";
// Refer to the DocNode documentation for more information
// on how to define a DocConfig
import { yourDocConfigs } from "./shared.ts";
import { fetchIdentity } from "./utils.ts";

export const { useDoc, usePresence, client } = createDocSyncClient({
  docBinding: DocNodeBinding([...yourDocConfigs]),
  server: { url: "http://localhost:8080", auth: { mode: "request" } },
  local: {
    provider: indexedDBProvider,
    getIdentity: async () => await fetchIdentity(),
  },
});
```

```ts tab="Vanilla"
// client.ts
import { DocSyncClient, indexedDBProvider } from "@docukit/docsync/client";
import { DocNodeBinding } from "@docukit/docsync/docnode";
// Refer to the DocNode documentation for more information
// on how to define a DocConfig
import { yourDocConfigs } from "./shared.ts";
import { fetchIdentity } from "./utils.ts";

export const client = new DocSyncClient({
  docBinding: DocNodeBinding([...yourDocConfigs]),
  server: { url: "http://localhost:8080", auth: { mode: "request" } },
  local: {
    provider: indexedDBProvider,
    getIdentity: async () => await fetchIdentity(),
  },
});
```

## 2. Server

```ts
// server.ts
import {
  DocSyncServer,
  inMemoryServerProvider,
} from "@docukit/docsync-react/server";
import { DocNodeBinding } from "@docukit/docsync-react/docnode";
// Refer to the DocNode documentation for more information
// on how to define a DocConfig
import { yourDocConfigs } from "./shared.ts";
import { getSessionFromCookie } from "./auth.ts";

// this will start the server on port 8080
const server = new DocSyncServer({
  docBinding: DocNodeBinding([...yourDocConfigs]),
  port: 8080,
  provider: inMemoryServerProvider(), // don't use in production
  authenticate: async ({ request }) => {
    const session = await getSessionFromCookie(request.headers);
    if (!session) return undefined;
    return { userId: session.user.id };
  },
});
```

For production, you'll want to bring your own storage provider — see [Providers](./providers).

## 3. Client Usage

```tsx tab="React"
// App.tsx
import { useDoc } from "./client";

const App = () => {
  const { doc } = useDoc({ type: "notes" });
  return <div>{doc.root.state.title.get()}</div>;
};
```
